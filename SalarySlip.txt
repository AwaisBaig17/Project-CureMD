using System;

namespace SalarySlip
{
    class Program
    {
        static void Main(string[] args)
        {
            int desig;
            Console.Write("Select your designation: \n 1-> Engineer \n 2-> Manager \n 3-> Analyst  ");
            Console.WriteLine();
            Console.Write("Your Selection: ")
            desig = Console.ReadLine();
            Console.WriteLine(" Gross salary is: "+CalcSalary(desig));

        }
        public static double CalcSalary(int desig)
        {
            double salary=0;
            switch (desig)
            {
                case 1:
                    Engineer myEng = new Engineer();
                    Salary = myEng.CalculateSalary();
                    break;

                case 2:
                    Manager myMng = new Manager();
                    Salary = myMng.CalculateSalary();
                    break;

                case 3:
                    Analyst myAnalyst = new Analyst();
                    Salary = myAnalyst.CalculateSalary();
                    break;

            }
            return Salary;
        }
    }
}
